117:     /**
118:      * Parses the raw xml data.

120:      * @since  0.3.1
121:      * @return void

123:     private function parseRawResponse() {
124:         try {
125:             $oSxmlResponse = new \SimpleXMLElement($this->sRawResponse);
126:         } catch (\Exception $oException) {
127:             $this->iReturnCode = 500;
128:             return self::RESPONSE_NOK;
129:         } // catch
130: 
131: 
132:         /* @var $oSxmlResponse SimpleXMLElement */
133:         if ($oSxmlResponse->success['code'] != null) {
134:             $this->iReturnCode = (int)$oSxmlResponse->success['code'];
135:             $this->iRemaining = (int)$oSxmlResponse->success['remaining'];
136:             $this->iResetDate = (int)$oSxmlResponse->success['resetdate'];
137: 
138:             if ($oSxmlResponse->retrieve != null) {
139:                 if ($oSxmlResponse->retrieve['token'] != null) {
140:                     $this->sToken = $oSxmlResponse->retrieve['token'];
141:                     $this->sTokenUrl = $oSxmlResponse->retrieve['url'];
142:                 } elseif ($oSxmlResponse->retrieve['apikey'] != null) {
143:                     $this->sApiKey = (string)$oSxmlResponse->retrieve['apikey'];
144:                 }
145:             }
146:             return self::RESPONSE_OK;
147:         } else {
148:             $this->iReturnCode = (int)$oSxmlResponse->error['code'];
149:             return self::RESPONSE_NOK;
150:         }
151:     }
